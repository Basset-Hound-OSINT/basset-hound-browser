## Export Formats Guide

### Overview

Basset Hound Browser v12.9.0 supports 8 export formats for different use cases:

| Format | Use Case | Size Efficiency | Best For |
|--------|----------|-----------------|----------|
| **JSON** | Structured data, replay-ready | Medium | General purpose, replay, logging |
| **CSV** | Tabular data, spreadsheets | High | Analysis, bulk processing |
| **HAR** | HTTP Archive, debugging | Medium | Network analysis, proxy/DevTools |
| **WARC** | Web Archive, compliance | Low | Long-term storage, legal holds |
| **SQLite** | Database, querying | High | Complex queries, relational data |
| **Markdown** | Human-readable reports | Medium | Documentation, manual review |
| **XML** | System integration, SOAP | Medium | Enterprise systems, workflows |
| **Custom** | User-defined templates | Variable | Specialized formats |

### 1. JSON Export

**Best For**: Replay scenarios, general-purpose logging, multi-agent coordination

**When to Use**:
- Saving session state for later replay
- Feeding data to external APIs
- Real-time multi-agent data sharing
- Long-term storage with compression

#### Basic Usage

```javascript
const client = new WebSocketClient('ws://localhost:8765');

// Export as JSON (formatted)
const result = await client.send({
  command: 'export_format_json',
  data_type: 'all',
  prettify: true,
  include_fields: ['url', 'status', 'headers', 'body', 'timestamp']
});

// Result: { success: true, data: {...}, stats: {...} }
console.log(result.data);
```

#### Advanced Options

```javascript
// Export with filters
await client.send({
  command: 'export_format_json',
  data_type: 'network_logs',
  filters: {
    statusCode: 200,
    resourceType: 'xhr',
    timeRange: { start: 1688000000000, end: 1688003600000 }
  },
  exclude_fields: ['base64Content', 'certificateChain'],
  output_path: '/data/exports/session-20260703.json'
});

// Export session data only
await client.send({
  command: 'export_format_json',
  data_type: 'session_data',
  prettify: true
});
```

**Output Structure**:
```json
{
  "networkLogs": [
    {
      "url": "https://api.example.com/data",
      "method": "GET",
      "status": 200,
      "headers": {...},
      "responseTime": 145,
      "timestamp": "2026-07-03T14:30:00Z"
    }
  ],
  "sessionData": {
    "cookies": [...],
    "localStorage": {...},
    "sessionStorage": {...}
  }
}
```

**File Size Optimization**:
- Exclude `base64Content` for 40-60% reduction
- Use `compress: true` in custom exports
- Prettify only for human review (adds 20-30%)

### 5. SQLite Export

**Best For**: Relational queries, complex analysis, embedded scenarios

**When to Use**:
- Complex SQL queries
- Joining with other data
- Building searchable indices
- Mobile app offline use

#### Basic Usage

```javascript
// Export to SQLite database
await client.send({
  command: 'export_format_sqlite',
  output_path: '/data/browser-session.db',
  tables: ['network_logs', 'cookies', 'local_storage', 'console_logs']
});
```

#### Querying Exported Data

```bash
# Open database
sqlite3 /data/browser-session.db

# Query network logs
SELECT url, status, responseTime 
FROM network_logs 
WHERE status >= 400 
ORDER BY responseTime DESC;

# Find slow requests
SELECT url, responseTime 
FROM network_logs 
WHERE responseTime > 1000;

# Cookie analysis
SELECT name, domain, value, expires 
FROM cookies 
WHERE domain LIKE '%.example.com';

# Cross-domain tracking (multi-agent analysis)
SELECT DISTINCT domain 
FROM cookies 
WHERE domain NOT IN (SELECT domain FROM first_party_domains);
```

#### Python Example

```python
import sqlite3
import json

conn = sqlite3.connect('/data/browser-session.db')
cursor = conn.cursor()

# Get all network logs
cursor.execute('SELECT * FROM network_logs')
logs = cursor.fetchall()

# Analyze by resource type
cursor.execute('''
  SELECT resourceType, COUNT(*) as count, AVG(responseTime) as avg_time
  FROM network_logs
  GROUP BY resourceType
  ORDER BY avg_time DESC
''')

for resource_type, count, avg_time in cursor.fetchall():
    print(f"{resource_type}: {count} requests, avg {avg_time}ms")

conn.close()
```

### Real-World Scenario: Multi-Target Investigation

```javascript
async function investigateMultipleTargets() {
  const targets = [
    'https://competitor-a.com',
    'https://competitor-b.com',
    'https://target.com'
  ];

  for (const target of targets) {
    // Create collaboration session per target
    const session = await client.send({
      command: 'start_collaboration_session',
      session_id: `investigation-${Date.now()}`,
      max_agents: 5
    });

    // Assign agents
    await Promise.all([
      client.send({
        command: 'assign_agent_to_target',
        session_id: session.session_id,
        agent_id: 'agent-tech',
        target_url: target
      }),
      client.send({
        command: 'assign_agent_to_target',
        session_id: session.session_id,
        agent_id: 'agent-competitor',
        target_url: target
      }),
      client.send({
        command: 'assign_agent_to_target',
        session_id: session.session_id,
        agent_id: 'agent-forensic',
        target_url: target
      })
    ]);

    // Wait for completion
    await new Promise(resolve => {
      const checkInterval = setInterval(async () => {
        const status = await client.send({
          command: 'get_collaboration_status',
          session_id: session.session_id
        });

        if (status.phase === 'complete') {
          clearInterval(checkInterval);
          resolve();
        }
      }, 5000);
    });

    // Export findings
    await client.send({
      command: 'export_collaboration_findings',
      session_id: session.session_id,
      format: 'json',
      output_path: `/reports/${target.split('//')[1]}-findings.json`
    });
  }
}
```

// Stream Basset findings to data warehouse
const EventEmitter = require('events');
const pg = require('pg');  // PostgreSQL

class BassetDataPipeline extends EventEmitter {
  constructor(dbConfig, bassetConfig) {
    super();
    this.pool = new pg.Pool(dbConfig);
    this.basset = new BassetClient(bassetConfig.url);
  }

  async start() {
    // Listen to Basset events
    this.basset.on('navigation', (event) => this.logNavigation(event));
    this.basset.on('network_request', (event) => this.logNetworkEvent(event));
    this.basset.on('detection', (event) => this.logDetection(event));
  }

  async logNavigation(event) {
    const client = await this.pool.connect();
    try {
      await client.query(
        'INSERT INTO navigation_log (url, timestamp, referrer, user_agent) VALUES ($1, $2, $3, $4)',
        [event.url, new Date(), event.referrer, event.userAgent]
      );
    } finally {
      client.release();
    }
  }

  async logNetworkEvent(event) {
    const client = await this.pool.connect();
    try {
      await client.query(
        `INSERT INTO network_log 
         (url, method, status, response_time, content_type, size, timestamp)
         VALUES ($1, $2, $3, $4, $5, $6, $7)`,
        [event.url, event.method, event.status, event.responseTime, 
         event.contentType, event.size, new Date()]
      );
    } finally {
      client.release();
    }
  }

  async logDetection(event) {
    const client = await this.pool.connect();
    try {
      await client.query(
        `INSERT INTO detection_log 
         (service, method, confidence, timestamp, response_data)
         VALUES ($1, $2, $3, $4, $5)`,
        [event.service, event.method, event.confidence, new Date(), 
         JSON.stringify(event.data)]
      );
    } finally {
      client.release();
    }
  }
}

### Integration Best Practices

#### 1. Error Handling & Retries

```javascript
async function robustIntegration(fn, maxRetries = 3, backoffMs = 1000) {
  let lastError;

  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error;
      if (attempt < maxRetries - 1) {
        const delay = backoffMs * Math.pow(2, attempt);
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
  }

  throw new Error(`Failed after ${maxRetries} attempts: ${lastError.message}`);
}

// Usage
const result = await robustIntegration(async () => {
  return await basset.send({
    command: 'navigate',
    url: 'https://protected-site.com'
  });
});
```

#### 2. Rate Limiting

```javascript
class RateLimitedBassetClient {
  constructor(bassetUrl, rateLimit = 10) {
    this.basset = new BassetClient(bassetUrl);
    this.queue = [];
    this.processing = false;
    this.rateLimit = rateLimit;  // requests per second
  }

  async send(command) {
    return new Promise((resolve, reject) => {
      this.queue.push({ command, resolve, reject });
      this.processQueue();
    });
  }

  async processQueue() {
    if (this.processing || this.queue.length === 0) return;

    this.processing = true;
    const startTime = Date.now();
    let processed = 0;

    while (this.queue.length > 0) {
      const { command, resolve, reject } = this.queue.shift();

      try {
        const result = await this.basset.send(command);
        resolve(result);
      } catch (error) {
        reject(error);
      }

      processed++;
      if (processed < this.rateLimit) {
        const elapsed = Date.now() - startTime;
        const targetTime = (processed / this.rateLimit) * 1000;
        if (elapsed < targetTime) {
          await new Promise(r => setTimeout(r, targetTime - elapsed));
        }
      }
    }

    this.processing = false;
  }
}
```
