## Quick Start: Integrating v12.9.0 API

### 5-Minute Setup

```javascript
const WebSocket = require('ws');

// 1. Connect
const token = 'your_auth_token';
const ws = new WebSocket(`ws://localhost:8765?token=${token}`);

// 2. Send command
ws.on('open', () => {
  ws.send(JSON.stringify({
    id: 1,
    command: 'navigate',
    params: { url: 'https://example.com' }
  }));
});

// 3. Handle response
ws.on('message', (data) => {
  const response = JSON.parse(data);
  if (response.success) {
    console.log('Command succeeded:', response.data);
  } else {
    console.error('Command failed:', response.error);
  }
});
```

## Export Commands Implementation

### Full Export Workflow

```javascript
class SessionExporter {
  constructor(wsClient) {
    this.wsClient = wsClient;
    this.requestId = 0;
  }

  async exportSession(options = {}) {
    const {
      format = 'json',
      includeNetwork = true,
      includeScreenshots = false,
      compression = 'gzip',
      outputDir = '/exports'
    } = options;

    const commands = [
      // Step 1: Get network logs
      includeNetwork && this.getNetworkLogs(),
      // Step 2: Get page snapshots
      this.getDOMSnapshots(),
      // Step 3: Export in requested format
      this.exportFormat(format, { outputDir, compression })
    ].filter(Boolean);

    return Promise.all(commands);
  }

  async getNetworkLogs() {
    return this.execute('export_format_json', {
      data_type: 'network_logs',
      prettify: true
    });
  }

  async getDOMSnapshots() {
    return this.execute('export_format_json', {
      data_type: 'session_data',
      include_fields: ['url', 'title', 'timestamp']
    });
  }

  async exportFormat(format, options) {
    const command = `export_format_${format}`;
    return this.execute(command, options);
  }

  async execute(command, params = {}) {
    return new Promise((resolve, reject) => {
      const id = ++this.requestId;
      const timeout = setTimeout(
        () => reject(new Error('Command timeout')),
        30000
      );

      const messageHandler = (data) => {
        const response = JSON.parse(data);
        if (response.id === id) {
          clearTimeout(timeout);
          this.wsClient.removeEventListener('message', messageHandler);
          
          if (response.success) {
            resolve(response.data);
          } else {
            reject(new Error(response.error.message));
          }
        }
      };

      this.wsClient.addEventListener('message', messageHandler);
      this.wsClient.send(JSON.stringify({
        id,
        command,
        params,
        timeout: 30000
      }));
    });
  }
}
```

```javascript
class CollaborativeSession {
  constructor(wsClient, userId) {
    this.wsClient = wsClient;
    this.userId = userId;
    this.sessionId = null;
    this.participants = new Map();
    this.annotations = [];
  }

  async createSession(name, options = {}) {
    const response = await this.execute('collaboration_create_session', {
      name,
      description: options.description || '',
      visibility: options.visibility || 'team',
      maxParticipants: options.maxParticipants || 10,
      features: {
        realTimeSharing: true,
        commandHistory: true,
        screenshotCapture: true,
        annotationTools: true,
        ...options.features
      }
    });

    this.sessionId = response.sessionId;
    console.log('Collaborative session created:', response.inviteLink);
    return response;
  }

  async addTeamMember(userId, role = 'contributor') {
    const response = await this.execute('collaboration_add_participant', {
      sessionId: this.sessionId,
      userId,
      role,
      permissions: this.getRolePermissions(role)
    });

    this.participants.set(userId, response);
    return response;
  }

  getRolePermissions(role) {
    const roleMap = {
      'moderator': ['view', 'annotate', 'comment', 'share', 'manage'],
      'contributor': ['view', 'annotate', 'comment'],
      'viewer': ['view', 'comment']
    };
    return roleMap[role] || roleMap.viewer;
  }

  async captureAndShare() {
    // Capture screenshot
    const screenshot = await this.execute('capture_screenshot_with_annotations', {
      type: 'viewport',
      format: 'png',
      quality: 'high'
    });

    // Share with team
    await this.execute('collaboration_share_screen', {
      sessionId: this.sessionId,
      quality: 'high',
      fps: 30
    });

    return screenshot;
  }

  async annotateIssue(screenshotId, issue) {
    const annotation = await this.execute('collaboration_create_annotation', {
      sessionId: this.sessionId,
      screenshotId,
      annotation: {
        type: 'rectangle',
        coordinates: issue.coordinates,
        color: issue.severity === 'high' ? '#FF0000' : '#FFFF00',
        label: issue.description,
        severity: issue.severity
      }
    });

    this.annotations.push(annotation);
    return annotation;
  }

  async getAuditLog(filter = {}) {
    return this.execute('collaboration_get_audit_log', {
      sessionId: this.sessionId,
      limit: filter.limit || 100,
      offset: filter.offset || 0,
      eventTypes: filter.eventTypes || null
    });
  }

  async execute(command, params) {
    return new Promise((resolve, reject) => {
      const id = Date.now();
      const messageHandler = (data) => {
        const response = JSON.parse(data);
        if (response.id === id) {
          this.wsClient.removeEventListener('message', messageHandler);
          if (response.success) {
            resolve(response.data);
          } else {
            reject(new Error(response.error.message));
          }
        }
      };

      this.wsClient.addEventListener('message', messageHandler);
      this.wsClient.send(JSON.stringify({
        id,
        command,
        params,
        timeout: 30000
      }));

      setTimeout(() => reject(new Error('Timeout')), 30000);
    });
  }
}

// Usage
const session = new CollaborativeSession(ws, 'user_123');
await session.createSession('E-commerce Checkout Testing');
await session.addTeamMember('user_456', 'contributor');
await session.addTeamMember('user_789', 'viewer');
const screenshot = await session.captureAndShare();
await session.annotateIssue(screenshot.id, {
  coordinates: { x: 100, y: 200, width: 300, height: 50 },
  description: 'Checkout button missing label',
  severity: 'high'
});
```

```javascript
class EvasionManager {
  constructor(wsClient) {
    this.wsClient = wsClient;
    this.fingerprintId = null;
    this.behavioralMode = null;

  async setupEvasion(targetSite = null) {
    // Step 1: Apply fingerprint
    const fingerprint = await this.applyFingerprint();

    const detections = await this.scanDetections(targetSite);

      fingerprint,
      behavioral,

      evasionScore: this.calculateEvasionScore()

    const response = await this.execute('apply_fingerprint_profile', {
      profileId: 'fp_enterprise_v2',

    const response = await this.execute('enable_behavioral_simulation', {
      mode: 'advanced',
      mouseMovement: {
        enabled: true,
        style: 'human-like',
        speed: 'variable'
      },
      typing: {
        enabled: true,
        typoRate: 0.02,
        pausesBetweenCharacters: true
      },
      scrolling: {
        enabled: true,
        pace: 'natural'
      },
      delays: {
        beforeInteraction: [100, 500],
        betweenCommands: [200, 1000],
        afterPageLoad: [1000, 3000]
      }

          break;
        case 'recaptcha':
          await this.adaptToRecaptcha(service);
          break;
        case 'datadome':
          await this.adaptToDataDome(service);
          break;
        // ... other services
      }

  }

  async adaptToRecaptcha(service) {

  }

  async adaptToDataDome(service) {
    // DataDome is behavior-focused - increase natural delays
    console.log('Adapting to DataDome detection');
  }

  async rotateFingerprint() {
    const response = await this.execute('rotate_fingerprint', {
      scope: 'session',
      preserveProfile: false,
      seedBase: 'user_specific'

    this.fingerprintId = response.newFingerprintId;

        }

      this.wsClient.addEventListener('message', handler);
      this.wsClient.send(JSON.stringify({ id, command, params }));

  }
}
```

## Reliability & Retry Implementation

### Automatic Retry Handler

```javascript
class ReliableCommandExecutor {
  constructor(wsClient, options = {}) {
    this.wsClient = wsClient;
    this.maxRetries = options.maxRetries || 3;
    this.baseBackoff = options.baseBackoff || 1000;
    this.maxBackoff = options.maxBackoff || 30000;
  }

  async execute(command, params = {}, options = {}) {
    const {
      retryable = true,
      timeout = 30000,
      onRetry = null
    } = options;

    let lastError = null;
    let attempts = 0;

    for (attempts = 1; attempts <= this.maxRetries; attempts++) {
      try {
        return await this.executeOnce(command, params, timeout);
      } catch (error) {
        lastError = error;

        // Check if error is retryable
        if (!retryable || !this.isRetryable(error) || attempts === this.maxRetries) {
          throw error;
        }

        // Calculate backoff
        const backoff = this.calculateBackoff(attempts);
        
        if (onRetry) {
          onRetry({
            attempt: attempts,
            error: error.message,
            nextRetryIn: backoff
          });
        }

        // Wait before retry
        await this.sleep(backoff);
      }
    }

    throw lastError;
  }

  async executeOnce(command, params, timeout) {
    return new Promise((resolve, reject) => {
      const id = Date.now();
      const timer = setTimeout(() => {
        this.wsClient.removeEventListener('message', handler);
        reject(new Error('COMMAND_TIMEOUT'));
      }, timeout);

      const handler = (data) => {
        const response = JSON.parse(data);
        if (response.id === id) {
          clearTimeout(timer);
          this.wsClient.removeEventListener('message', handler);

          if (response.success) {
            resolve(response.data);
          } else {
            const error = new Error(response.error.message);
            error.code = response.error.code;
            error.details = response.error.details;
            reject(error);
          }
        }
      };

      this.wsClient.addEventListener('message', handler);
      this.wsClient.send(JSON.stringify({
        id,
        command,
        params,
        timeout
      }));
    });
  }

  isRetryable(error) {
    const transientErrors = [
      'ETIMEDOUT',
      'ECONNRESET',
      'ECONNREFUSED',
      'COMMAND_TIMEOUT',
      'RESOURCE_UNAVAILABLE'
    ];

    return transientErrors.includes(error.code || error.message);
  }

  calculateBackoff(attempt) {
    const exponential = Math.pow(2, attempt - 1) * this.baseBackoff;
    const jitter = Math.random() * exponential * 0.1;
    return Math.min(exponential + jitter, this.maxBackoff);
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// Usage
const executor = new ReliableCommandExecutor(ws, {
  maxRetries: 3,
  baseBackoff: 1000
});

try {
  const result = await executor.execute('navigate', {
    url: 'https://example.com'
  }, {
    retryable: true,
    timeout: 30000,
    onRetry: (info) => {
      console.log(`Retrying after ${info.nextRetryIn}ms (attempt ${info.attempt})`);
    }
  });
  console.log('Success:', result);
} catch (error) {
  console.error('Failed after retries:', error.message);
}
```

## Health Monitoring Implementation

### Health Check Wrapper

```javascript
class HealthMonitor {
  constructor(httpClient, wsClient) {
    this.httpClient = httpClient;
    this.wsClient = wsClient;
    this.lastHealthCheck = null;
    this.healthCheckInterval = 30000; // 30 seconds
  }

  async startMonitoring() {
    setInterval(() => this.checkHealth(), this.healthCheckInterval);
  }

  async checkHealth() {
    try {
      const health = await this.getDetailedHealth();
      this.lastHealthCheck = health;
      this.analyzeHealth(health);
      return health;
    } catch (error) {
      console.error('Health check failed:', error.message);
      return null;
    }
  }

  async getDetailedHealth() {
    const response = await this.httpClient.get('http://localhost:8080/health/detailed');
    return response.json();
  }

  analyzeHealth(health) {
    // Check memory
    const memoryPercent = health.components.memory.percentage;
    if (memoryPercent > 80) {
      console.warn(`High memory usage: ${memoryPercent}%`);
    }

    // Check CPU
    const cpuUsage = health.components.cpu.usage;
    if (cpuUsage > 75) {
      console.warn(`High CPU usage: ${cpuUsage}%`);
    }

    // Check connections
    const connections = health.components.websocket.activeConnections;
    const maxConnections = health.components.websocket.maxConnections;
    if (connections / maxConnections > 0.8) {
      console.warn(`Connection pool near capacity: ${connections}/${maxConnections}`);
    }
  }

  async getReliabilityMetrics() {
    return new Promise((resolve, reject) => {
      const id = Date.now();
      const handler = (data) => {
        const response = JSON.parse(data);
        if (response.id === id) {
          this.wsClient.removeEventListener('message', handler);
          if (response.success) resolve(response.data);
          else reject(new Error(response.error.message));
        }
      };

      this.wsClient.addEventListener('message', handler);
      this.wsClient.send(JSON.stringify({
        id,
        command: 'get_reliability_metrics',
        params: { timeWindow: 3600000, detailed: true }
      }));

      setTimeout(() => reject(new Error('Timeout')), 10000);
    });
  }
}

// Usage
const monitor = new HealthMonitor(fetch, ws);
await monitor.startMonitoring();
const metrics = await monitor.getReliabilityMetrics();
console.log('Reliability:', metrics.slaCompliance.actual + '%');
```

## Best Practices

### 1. Always Implement Retry Logic

```javascript
// Good
const result = await executor.execute(command, params, { retryable: true });

// Avoid
const result = await executor.executeOnce(command, params);
```

### 2. Monitor Health Proactively

```javascript
// Start health monitoring
const monitor = new HealthMonitor(fetch, ws);
await monitor.startMonitoring();
```

### 3. Use Appropriate Timeouts

```javascript
// Dynamic timeout based on command type
const timeouts = {
  navigate: 30000,
  screenshot: 15000,
  click: 5000,
  fill: 5000
};

await executor.execute(command, params, {
  timeout: timeouts[command]
});
```

### 4. Implement Graceful Degradation

```javascript
try {
  const premium = await exportJSON({ compression: 'gzip' });
} catch {
  // Fallback to no compression
  const basic = await exportJSON({ compression: null });
}
```

### 5. Log and Monitor Errors

```javascript
executor.execute(command, params, {
  onRetry: (info) => {
    logger.warn('Command retry', {
      command,
      attempt: info.attempt,
      reason: info.error
    });
  }
});
```

## Performance Optimization

### Connection Pooling

```javascript
class ConnectionPool {
  constructor(size = 5) {
    this.size = size;
    this.connections = [];
    this.available = [];
  }

  async initialize() {
    for (let i = 0; i < this.size; i++) {
      const ws = await this.createConnection();
      this.connections.push(ws);
      this.available.push(ws);
    }
  }

  async acquire() {
    while (this.available.length === 0) {
      await new Promise(r => setTimeout(r, 100));
    }
    return this.available.pop();
  }

  release(ws) {
    this.available.push(ws);
  }
}
```
