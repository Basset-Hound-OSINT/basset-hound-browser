```javascript
/**
 * WebSocket Commands for Monitoring & Metrics (v12.7.0 Feature 4)
 * Wrapper for existing monitoring infrastructure in src/monitoring/websocket-handlers.js
 */

function registerMonitoringMetricsCommands(server, mainWindow) {
  // Lazy initialization of metrics stack if not already done
  if (!server.metricsCollector) {
    const { MetricsCollector } = require('../../src/monitoring/metrics-collector');
    const { MetricsAggregator } = require('../../src/monitoring/metrics-aggregator');
    const { AlertManager } = require('../../src/monitoring/alert-manager');
    const { MetricsStore } = require('../../src/monitoring/metrics-store');
    
    server.metricsCollector = new MetricsCollector();
    server.metricsAggregator = new MetricsAggregator();
    server.alertManager = new AlertManager();
    server.metricsStore = new MetricsStore();
  }
  
  // Delegate to existing monitoring handlers
  const { registerMonitoringHandlers } = require('../../src/monitoring/websocket-handlers');
  registerMonitoringHandlers(
    server.commandDispatcher,
    server.metricsCollector,
    server.metricsAggregator,
    server.alertManager,
    server.metricsStore,
    server
  );
}

module.exports = { registerMonitoringMetricsCommands };
```
