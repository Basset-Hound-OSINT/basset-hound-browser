const APIGateway = require('./src/api/gateway');

const gateway = new APIGateway({ port: 8765 });

// Register services
gateway.registerService('user-service', {
  host: 'user-svc',
  port: 3001
});

// Register routes
gateway.registerRoute('/users', 'GET', {
  service: 'user-service',
  cache: true,
  cacheTTL: 300000
});

// Handle requests
const response = await gateway.handleRequest({
  path: '/users',
  method: 'GET',
  headers: {}
});

SERVICE REGISTRY EXAMPLE:
───────────────────────

const ServiceRegistry = require('./src/api/service-registry');

const registry = new ServiceRegistry();

// Register instance
registry.registerInstance('user-service', 'instance-1', {
  host: 'localhost',
  port: 3001
});

// Discover instances
const instances = registry.getInstances('user-service');
const instance = registry.getInstance('user-service', 'round-robin');

// Start health checking
registry.startHealthChecking();

MESH CONTROLLER EXAMPLE:
──────────────────────

const ServiceMeshController = require('./src/mesh/mesh-controller');

const mesh = new ServiceMeshController();

// Create virtual service
mesh.createVirtualService('user-service', {
  hosts: ['user-service']
});

// Configure routing
mesh.configureRouting('user-service', [{
  name: 'route-1',
  destinations: [{ host: 'user-service' }]
}]);

// Setup mTLS
mesh.setupPeerAuthentication('user-service-mtls', {
  mtls: { mode: 'STRICT' }
});

TRACER EXAMPLE:
──────────────

const DistributedTracer = require('./src/observability/tracer');

const tracer = new DistributedTracer({ serviceName: 'api' });

// Start trace
const { traceId } = tracer.startTrace();

// Start span
const { spanId } = tracer.startSpan(traceId, 'get-user');

// Add tags
tracer.addSpanTag(spanId, 'http.method', 'GET');
tracer.addSpanTag(spanId, 'http.status_code', 200);

// End span
tracer.endSpan(spanId);

// Export
tracer.startPeriodicExport();

METRICS EXAMPLE:
───────────────

const MetricsAggregator = require('./src/observability/metrics');

const metrics = new MetricsAggregator();

// Register metrics

const gauge = metrics.registerGauge('cpu_usage');
const histogram = metrics.registerHistogram('latency');

// Record metrics

histogram.observe(0.42);

// Export
