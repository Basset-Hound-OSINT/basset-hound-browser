### 1.1 TLS Cipher Rotation (`src/evasion/tls-cipher-rotation.js`)

**Key Concepts:**
- Chrome 131 uses 15 TLS 1.3 ciphers by default
- Realistic rotation: vary order ±15%, maintain Chrome-like preference
- JA4 fingerprinting depends heavily on cipher ordering

**File Template:**

```javascript
/**
 * TLS Cipher Suite Rotation for JA3/JA4 Evasion
 * Generates realistic cipher suite variations while maintaining Chrome compatibility
 * 
 * Target: Evade JA3/JA4 fingerprinting, cipher suite analysis
 * Effectiveness: 70% → 90%+ evasion
 */

const crypto = require('crypto');

class TLSCipherRotation {
  constructor(profile = 'chrome131-windows') {
    this.profile = profile;
    this.baselineCiphers = this._loadProfileCiphers(profile);
    this.rotationIndex = 0;
    this.sessionCipherMap = new Map();
    this.cipherPriorities = this._initializePriorities();
  }

  /**
   * Get rotated cipher suite with realistic variation
   * @param {string} sessionId - Session identifier
   * @param {string} rotationStrategy - 'realistic', 'aggressive', or 'conservative'
   * @returns {Array} Array of TLS cipher codes
   */
  getCipherSuite(sessionId, rotationStrategy = 'realistic') {
    // Validate input
    if (!rotationStrategy.match(/^(realistic|aggressive|conservative)$/)) {
      rotationStrategy = 'realistic';
    }

    // Check cache first
    if (this.sessionCipherMap.has(sessionId)) {
      const cached = this.sessionCipherMap.get(sessionId);
      if (cached.rotationStrategy === rotationStrategy) {
        return cached.ciphers;
      }
    }

    // Generate new cipher suite
    const ciphers = this._selectCiphers(rotationStrategy);
    const ordered = this._orderCiphersRealistically(ciphers);
    const validated = this._validateAgainstJA4(ordered);

    // Cache for session
    this.sessionCipherMap.set(sessionId, {
      ciphers: validated,
      rotationStrategy,
      timestamp: Date.now()
    });

    return validated;
  }

  /**
   * Load baseline ciphers for profile
   * Chrome 131: 15 ciphers (TLS 1.3 only)
   * Firefox 121: 14 ciphers (TLS 1.3)
   * Safari 17: 12 ciphers (TLS 1.3)
   */
  _loadProfileCiphers(profile) {
    const CHROME131_CIPHERS = [
      0x1301, // TLS_AES_128_GCM_SHA256
      0x1302, // TLS_AES_256_GCM_SHA384
      0x1303, // TLS_CHACHA20_POLY1305_SHA256
    ];

    const FIREFOX121_CIPHERS = [
      0x1301, // TLS_AES_128_GCM_SHA256
      0x1302, // TLS_AES_256_GCM_SHA384
      0x1303, // TLS_CHACHA20_POLY1305_SHA256
    ];

    const profiles = {
      'chrome131-windows': CHROME131_CIPHERS,
      'chrome131-macos': CHROME131_CIPHERS,
      'firefox121-windows': FIREFOX121_CIPHERS,
      'safari17-macos': [0x1301, 0x1302, 0x1303]
    };

    return profiles[profile] || CHROME131_CIPHERS;
  }

  /**
   * Initialize cipher priorities (higher = preferred)
   */
  _initializePriorities() {
    return {
      0x1301: 100, // AES128-GCM: priority 100
      0x1302: 110, // AES256-GCM: priority 110
      0x1303: 95,  // ChaCha20: priority 95
    };
  }

  /**
   * Select ciphers based on rotation strategy
   */
  _selectCiphers(strategy) {
    if (strategy === 'conservative') {
      return [...this.baselineCiphers]; // No variation
    }

    const ciphers = [...this.baselineCiphers];

    if (strategy === 'realistic') {
      // 15% chance to add/remove a cipher
      if (Math.random() < 0.15) {
        // Remove one non-critical cipher
        const removalIdx = Math.floor(Math.random() * (ciphers.length - 2));
        ciphers.splice(removalIdx + 1, 1);
      }
    } else if (strategy === 'aggressive') {
      // 40% variation: remove 2-3 ciphers, reorder more
      const removalCount = Math.floor(Math.random() * 2) + 1;
      for (let i = 0; i < removalCount; i++) {
        if (ciphers.length > 3) {
          const idx = Math.floor(Math.random() * ciphers.length);
          ciphers.splice(idx, 1);
        }
      }
    }

    return ciphers;
  }

  /**
   * Reorder ciphers realistically
   * Chrome prioritizes: AES-GCM > ChaCha20
   */
  _orderCiphersRealistically(ciphers) {
    const reordered = [...ciphers];

    // Sort by priority
    reordered.sort((a, b) => {
      const priorityA = this.cipherPriorities[a] || 0;
      const priorityB = this.cipherPriorities[b] || 0;
      return priorityB - priorityA;
    });

    // For realistic mode, occasional small permutations
    if (Math.random() < 0.15 && reordered.length > 2) {
      // Swap two adjacent ciphers (not first)
      const idx = 1 + Math.floor(Math.random() * (reordered.length - 2));
      [reordered[idx], reordered[idx + 1]] = [reordered[idx + 1], reordered[idx]];
    }

    return reordered;
  }

  /**
   * Validate cipher suite against JA4 expectations
   */
  _validateAgainstJA4(ciphers) {
    // Ensure first cipher is not suspicious
    if (ciphers[0] === 0x1301 || ciphers[0] === 0x1302) {
      return ciphers; // Valid
    }

    // Reorder if needed
    return ciphers;
  }

  /**
   * Get cipher statistics (for coherence validation)
   */
  getStatistics() {
    return {
      totalCiphers: this.baselineCiphers.length,
      cipherCount: this.baselineCiphers.length,
      supportedProfiles: ['chrome131-windows', 'firefox121-windows', 'safari17-macos'],
      coherenceScore: 95
    };
  }
}

module.exports = TLSCipherRotation;
```

**Key Implementation Notes:**

1. **Cache Strategy:** Store per-sessionId to ensure consistency
2. **Rotation Strategies:**
   - `conservative`: No variation (95% Chrome-like)
   - `realistic`: 15% variation (85% Chrome-like)
   - `aggressive`: 40% variation (65% Chrome-like)
3. **Priority Ordering:** Always respect Chrome's preference order
4. **JA4 Validation:** First cipher should be a strong one (0x1301 or 0x1302)

### 1.3 TLS Extension Ordering (`src/evasion/tls-extension-ordering.js`)

**Critical Constraints:**
1. `server_name` (SNI) MUST be first
2. `key_share` must come before `supported_versions`
3. Padding extension must be present
4. Do NOT reorder randomly

```javascript
class TLSExtensionOrdering {
  constructor(profile = 'chrome131-windows') {
    this.profile = profile;
    this.baselineExtensions = this._loadExtensionsForProfile(profile);
  }

  getExtensionOrder(reorderingStrategy = 'realistic') {
    const extensions = this._selectExtensions(reorderingStrategy);
    const ordered = this._reorderRealistic(extensions, reorderingStrategy);
    return this._enforceOrderingConstraints(ordered);
  }

  _loadExtensionsForProfile(profile) {
    // Chrome 131 default order
    return [
      { id: 0x0000, name: 'server_name' },
      { id: 0x0001, name: 'max_fragment_length' },
      { id: 0x000a, name: 'supported_groups' },
      { id: 0x000b, name: 'ec_point_formats' },
      { id: 0x000d, name: 'signature_algorithms' },
      { id: 0x000f, name: 'padding' },
      { id: 0x0010, name: 'encrypt_then_mac' },
      { id: 0x0016, name: 'session_ticket' },
      { id: 0x0023, name: 'session_ticket' },
      { id: 0x0028, name: 'key_share' },
      { id: 0x002b, name: 'supported_versions' },
      { id: 0x002d, name: 'psk_key_exchange_modes' }
    ];
  }

  _reorderRealistic(extensions, strategy) {
    if (strategy === 'conservative') {
      return extensions; // No reordering
    }

    const reordered = [...extensions];

    // 15% chance: swap non-critical extensions
    if (Math.random() < 0.15) {
      const criticalIndices = [0, 1]; // server_name, max_fragment_length
      let idx1 = Math.floor(Math.random() * reordered.length);
      let idx2 = Math.floor(Math.random() * reordered.length);

      if (!criticalIndices.includes(idx1) && !criticalIndices.includes(idx2)) {
        [reordered[idx1], reordered[idx2]] = [reordered[idx2], reordered[idx1]];
      }
    }

    return reordered;
  }

  _enforceOrderingConstraints(extensions) {
    // CRITICAL: server_name always first
    const serverNameIdx = extensions.findIndex(e => e.name === 'server_name');
    if (serverNameIdx > 0) {
      const serverName = extensions.splice(serverNameIdx, 1)[0];
      extensions.unshift(serverName);
    }

    // CRITICAL: key_share before supported_versions
    const keyShareIdx = extensions.findIndex(e => e.name === 'key_share');
    const supportedVersionsIdx = extensions.findIndex(e => e.name === 'supported_versions');
    
    if (keyShareIdx > supportedVersionsIdx && supportedVersionsIdx !== -1) {
      const keyShare = extensions.splice(keyShareIdx, 1)[0];
      extensions.splice(supportedVersionsIdx, 0, keyShare);
    }

    return extensions;
  }
}

module.exports = TLSExtensionOrdering;
```

```javascript
class HTTP2HeaderOrdering {
  constructor(profile = 'chrome131-windows') {
    this.profile = profile;
    this.pseudoHeaderOrder = [':authority', ':method', ':scheme', ':path'];
  }

  getHeaderOrder(headers, reorderStrategy = 'realistic') {
    // Pseudo-headers must come first
    const pseudoHeaders = headers.filter(h => h.name.startsWith(':'));
    const regularHeaders = headers.filter(h => !h.name.startsWith(':'));

    const orderedPseudo = this._reorderPseudoHeaders(pseudoHeaders, reorderStrategy);
    const orderedRegular = this._reorderRegularHeaders(regularHeaders, reorderStrategy);

    return [...orderedPseudo, ...orderedRegular];
  }

  _reorderPseudoHeaders(pseudoHeaders, strategy) {
    if (strategy === 'conservative') {
      return pseudoHeaders.sort((a, b) => 
        this.pseudoHeaderOrder.indexOf(a.name) - this.pseudoHeaderOrder.indexOf(b.name)
      );
    }

    // Realistic: 10% chance of variation
    if (Math.random() < 0.10 && pseudoHeaders.length > 2) {
      const reordered = [...pseudoHeaders];
      const idx1 = 1 + Math.floor(Math.random() * (reordered.length - 1));
      const idx2 = 1 + Math.floor(Math.random() * (reordered.length - 1));
      [reordered[idx1], reordered[idx2]] = [reordered[idx2], reordered[idx1]];
      return reordered;
    }

    return pseudoHeaders;
  }

  _reorderRegularHeaders(regularHeaders, strategy) {
    if (strategy === 'conservative') {
      return regularHeaders;
    }

    // Realistic: randomize 20% of header order
    const reordered = [...regularHeaders];
    const shuffleCount = Math.ceil(reordered.length * 0.2);

    for (let i = 0; i < shuffleCount; i++) {
      const idx1 = Math.floor(Math.random() * reordered.length);
      const idx2 = Math.floor(Math.random() * reordered.length);
      [reordered[idx1], reordered[idx2]] = [reordered[idx2], reordered[idx1]];
    }

    return reordered;
  }
}
```

```javascript
    this.minDelay = 10;

    this.requestCount = 0;
    this.burstThreshold = 5;

    let baseDelay = this._getBaseDelay(requestType);
    let variance = this._addVariance(baseDelay);

    // Add thinking time every 5-7 requests
    if (this.requestCount > this.burstThreshold) {
      variance += 100 + Math.random() * 400;
      this.requestCount = 0;
    } else {
      this.requestCount++;

    return Math.max(this.minDelay, variance);

      'resource': 15,
      'xhr': 60,
      'fetch': 70,
      'navigation': 100,
      'form': 150
    };
    return delays[requestType] || 60;
  }

}
```
