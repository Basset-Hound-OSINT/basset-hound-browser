```javascript
class TimeoutProtection {
  // Wrap async operations with timeout
  async withTimeout(promise, timeoutMs, operationName) {
    return Promise.race([
      promise,
      new Promise((_, reject) =>
        setTimeout(() => reject(
          new Error(`${operationName} timed out after ${timeoutMs}ms`)
        ), timeoutMs)
      )
    ]);
  }
  
  // JavaScript execution with execSync timeout + fallback
  executeJavaScriptSafe(code) {
    try {
      return this.withTimeout(
        this.executeViaExecSync(code),
        5000,
        'JavaScript execution'
      );
    } catch (error) {
      if (error.message.includes('timed out')) {
        // Fallback to node-forge or cached result
        return this.executeViaFallback(code);
      }
      throw error;
    }
  }
  
  // Command wrapper to add timeout protection
  wrapCommand(handler, timeoutMs) {
    return async (command) => {
      try {
        return await this.withTimeout(
          handler(command),
          timeoutMs,
          `Command: ${command.command}`
        );
      } catch (error) {
        return { error: `Command timed out: ${error.message}` };
      }
    };
  }
}
```

```javascript
    try {
      const handler = this.handlers[command.command];
      if (!handler) {
        return { error: 'Unknown command' };
      }
      
      // Wrap handler with promise error handling
      return await this.withErrorHandling(
        handler.call(this, command)

      return await Promise.resolve(promise);
    } catch (error) {
      logger.error('Promise rejection:', error);
      throw error; // Re-throw for command dispatcher to catch
    }
  }
}
```

**Problem:** Double-execution of commands due to race conditions

**Solution - IPC Safety Lock:**
```javascript
class IPCSafety {
  constructor() {
    this.pendingOperations = new Map(); // Command ID → Promise
    this.lastExecuted = new Map();      // Command ID → timestamp
  }
  
  // Check if command already executing
  async executeOncePerSession(commandId, handler) {
    // If already executing, wait for existing
    if (this.pendingOperations.has(commandId)) {
      return this.pendingOperations.get(commandId);
    }
    
    // If recently executed, return cached result
    if (this.lastExecuted.has(commandId)) {
      const lastTime = this.lastExecuted.get(commandId);
      if (Date.now() - lastTime < 100) { // Deduplicate within 100ms
        return this.getCachedResult(commandId);
      }
    }
    
    // Execute and cache
    const promise = handler();
    this.pendingOperations.set(commandId, promise);
    
    try {
      const result = await promise;
      this.lastExecuted.set(commandId, Date.now());
      this.cacheResult(commandId, result);
      return result;
    } finally {
      this.pendingOperations.delete(commandId);
    }
  }
}
```

#### 4.1 Message Throughput Optimization - 6 hours

**Current Baseline (v12.4.0):**

- 50 concurrent: 400-450 msg/sec
- 100 concurrent: 450-500 msg/sec
- Target: 500+ msg/sec maintained

**Optimization Strategies:**

1. **Message Queue Batching**
   - Buffer rapid commands (< 10ms apart)
   - Execute batch as single transaction
   - Reduce per-command overhead
   - Expected: 10-15% improvement

2. **Connection Pooling**
   - Reuse WebSocket connections
   - Pre-allocated buffer pool
   - Reduce connection setup time
   - Expected: 5-10% improvement

3. **Asynchronous Command Dispatch**
   - Non-blocking queue processing
   - Parallel command execution where safe
   - Expected: 15-20% improvement

4. **Memory Efficiency**
   - Object pooling (reuse command objects)
   - Reduce GC pressure
   - Expected: 5-8% improvement

**Implementation:**
```javascript
class OptimizedMessageDispatcher {
  constructor() {
    this.commandQueue = [];
    this.batchWindow = 5; // ms
    this.processing = false;
    this.objectPool = new ObjectPool(1000);
  }
  
  // Queue commands with batching
  queueCommand(command) {
    this.commandQueue.push(command);
    
    // Schedule batch processing
    if (!this.processing) {
      setTimeout(() => this.processBatch(), this.batchWindow);
    }
  }
  
  // Process queued commands as batch
  async processBatch() {
    if (this.commandQueue.length === 0) return;
    
    this.processing = true;
    const batch = this.commandQueue.splice(0);
    
    // Parallel processing for independent commands
    const results = await Promise.all(
      batch.map(cmd => this.executeCommand(cmd))
    );
    
    this.processing = false;
    return results;
  }
  
  // Command execution with object pooling
  async executeCommand(command) {
    const context = this.objectPool.acquire();
    try {
      return await this.dispatch(command, context);
    } finally {
      this.objectPool.release(context);
    }
  }
}

class ObjectPool {
  constructor(size) {
    this.pool = Array(size).fill().map(() => ({}));
    this.available = this.pool.slice();
  }
  
  acquire() {
    return this.available.pop() || {};
  }
  
  release(obj) {
    Object.keys(obj).forEach(k => delete obj[k]);
    this.available.push(obj);
  }
}
```
