class EmailNotifier {
  constructor(smtpConfig) {
    // smtpConfig: {host, port, user, password, from}
    this.mailer = createTransporter(smtpConfig)
  }
  
  async sendAlert(alert, recipients) {
    const html = this.formatAlertHTML(alert)
    return this.mailer.sendMail({
      from: this.config.from,
      to: recipients.join(','),
      subject: `[${alert.severity}] ${alert.metric} Alert`,
      html: html
    })
  }
  
  async sendDailyDigest(metrics, recipients) {
    // Send daily summary email
    const html = this.formatDigestHTML(metrics)

      from: this.config.from,
      to: recipients.join(','),
      subject: `Daily Metrics Digest - ${new Date().toDateString()}`,
      html: html
    })
