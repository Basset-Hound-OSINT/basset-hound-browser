## EXECUTIVE SUMMARY

The Distributed Browser Pool (Feature 3) enables Basset Hound Browser to manage multiple browser instances as a scalable resource pool, distributing workloads across instances and maintaining session coherence during failover. This feature is critical for high-volume automation scenarios (100+ concurrent requests) and multi-region deployments.

**Key Class: BrowserPoolManager**

```javascript
class BrowserPoolManager {
  constructor(options = {}) {
    // Configuration
    this.minInstances = options.minInstances || 1;      // Minimum pool size
    this.maxInstances = options.maxInstances || 10;     // Maximum pool size
    this.instanceTimeout = options.instanceTimeout || 600000; // 10 min idle timeout
    this.healthCheckInterval = options.healthCheckInterval || 30000; // 30 sec
    this.spinUpTimeout = options.spinUpTimeout || 5000; // 5 sec to start instance
    
    // Storage
    this.instances = new Map();        // instanceId → InstanceState
    this.loadBalancer = null;          // LoadBalancer instance
    this.resourceMonitor = null;       // ResourceMonitor instance
    this.sessionCoordinator = null;    // Distributed session coordinator
    
    // Metrics
    this.metrics = {
      totalSpinUps: 0,
      totalSpinDowns: 0,
      totalFailures: 0,
      averageSpinUpTime: 0,
      peakConcurrentInstances: 0,
      uptime: 0,
      lastUpdated: Date.now()
    };
  }

  /**
   * Initialize pool with minimum instances
   * @returns {Promise<PoolStatus>}
   */
  async initialize() {
    console.log(`[BrowserPoolManager] Initializing pool (min: ${this.minInstances})`);
    
    try {
      // Create load balancer
      this.loadBalancer = new LoadBalancer(this);
      
      // Create resource monitor
      this.resourceMonitor = new ResourceMonitor(this);
      
      // Spin up minimum instances
      for (let i = 0; i < this.minInstances; i++) {
        await this.spinUpInstance({
          isolated: true,
          prewarmed: true
        });
      }
      
      // Start health check loop
      this._startHealthCheck();
      
      return {
        success: true,
        instanceCount: this.instances.size,
        status: 'initialized'
      };
    } catch (error) {
      console.error(`[BrowserPoolManager] Init failed: ${error.message}`);
      return { success: false, error: error.message };
    }
  }

  /**
   * Spin up new browser instance
   * @param {Object} config - Instance configuration
   * @returns {Promise<string>} Instance ID
   */
  async spinUpInstance(config = {}) {
    if (this.instances.size >= this.maxInstances) {
      throw new Error(`Pool at max capacity (${this.maxInstances})`);
    }
    
    const startTime = Date.now();
    const instanceId = `instance-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    
    try {
      // Create instance state
      const instanceState = {
        id: instanceId,
        status: 'spinning-up',
        port: 8765 + this.instances.size,
        pid: null,
        startTime: startTime,
        lastHeartbeat: startTime,
        connections: 0,
        maxConnections: config.maxConnections || 200,
        cpuUsage: 0,
        memoryUsage: 0,
        metrics: {}
      };
      
      this.instances.set(instanceId, instanceState);
      
      // Start WebSocket server on instance port
      await this._startInstanceServer(instanceId, config);
      
      instanceState.status = 'healthy';
      instanceState.lastHeartbeat = Date.now();
      
      // Track spin up time
      const spinUpTime = Date.now() - startTime;
      this._updateSpinUpMetric(spinUpTime);
      
      this.metrics.totalSpinUps++;
      
      console.log(`[BrowserPoolManager] Instance ${instanceId} ready in ${spinUpTime}ms`);
      
      return instanceId;
    } catch (error) {
      this.instances.delete(instanceId);
      this.metrics.totalFailures++;
      throw error;
    }
  }

  /**
   * Spin down browser instance
   * @param {string} instanceId - Instance to shut down
   * @returns {Promise<void>}
   */
  async spinDownInstance(instanceId) {
    const instance = this.instances.get(instanceId);
    if (!instance) return;
    
    try {
      instance.status = 'shutting-down';
      
      // Close all active connections
      await this._closeInstanceConnections(instanceId);
      
      // Stop WebSocket server
      await this._stopInstanceServer(instanceId);
      
      // Remove from pool
      this.instances.delete(instanceId);
      this.metrics.totalSpinDowns++;
      
      console.log(`[BrowserPoolManager] Instance ${instanceId} shut down`);
    } catch (error) {
      console.error(`[BrowserPoolManager] Error shutting down ${instanceId}: ${error.message}`);
    }
  }

  /**
   * Get instance state
   * @param {string} instanceId - Instance to query
   * @returns {InstanceState}
   */
  getInstanceState(instanceId) {
    return this.instances.get(instanceId);
  }

  /**
   * List all instances
   * @returns {InstanceState[]}
   */
  listInstances() {
    return Array.from(this.instances.values());
  }

  /**
   * Monitor health and auto-cleanup
   * @private
   */
  _startHealthCheck() {
    this.healthCheckTimer = setInterval(() => {
      const now = Date.now();
      
      for (const [instanceId, instance] of this.instances) {
        // Check heartbeat
        const timeSinceHeartbeat = now - instance.lastHeartbeat;
        
        if (timeSinceHeartbeat > 60000) { // 60 sec timeout
          console.warn(`[BrowserPoolManager] Instance ${instanceId} unresponsive, removing`);
          this.spinDownInstance(instanceId);
        }
      }
      
      // Auto-scale based on resource usage
      this._evaluateAutoScale();
    }, this.healthCheckInterval);
  }

  /**
   * Evaluate and perform auto-scaling
   * @private
   */
  async _evaluateAutoScale() {
    const totalConnections = Array.from(this.instances.values())
      .reduce((sum, inst) => sum + inst.connections, 0);
    
    const avgConnectionsPerInstance = totalConnections / this.instances.size;
    
    // Scale up if average > 150 connections per instance
    if (avgConnectionsPerInstance > 150 && this.instances.size < this.maxInstances) {
      console.log(`[BrowserPoolManager] Scaling up (${avgConnectionsPerInstance.toFixed(0)} avg conn/instance)`);
      await this.spinUpInstance();
    }
    
    // Scale down if average < 20 and we have excess capacity
    if (avgConnectionsPerInstance < 20 && this.instances.size > this.minInstances) {
      console.log(`[BrowserPoolManager] Scaling down (${avgConnectionsPerInstance.toFixed(0)} avg conn/instance)`);
      const idleInstance = this._findIdleInstance();
      if (idleInstance) {
        await this.spinDownInstance(idleInstance.id);
      }
    }
  }

  /**
   * Find least busy instance
   * @private
   */
  _findIdleInstance() {
    let minConnections = Infinity;
    let idleInstance = null;
    
    for (const instance of this.instances.values()) {
      if (instance.connections < minConnections && instance.status === 'healthy') {
        minConnections = instance.connections;
        idleInstance = instance;
      }
    }
    
    return idleInstance;
  }

  /**
   * Update spin-up time metric (exponential moving average)
   * @private
   */
  _updateSpinUpMetric(newTime) {
    const alpha = 0.3; // EMA smoothing factor
    this.metrics.averageSpinUpTime = 
      alpha * newTime + (1 - alpha) * this.metrics.averageSpinUpTime;
  }

  /**
   * Get pool metrics and status
   * @returns {PoolMetrics}
   */
  getMetrics() {
    return {
      ...this.metrics,
      instances: this.instances.size,
      minInstances: this.minInstances,
      maxInstances: this.maxInstances,
      totalConnections: Array.from(this.instances.values())
        .reduce((sum, inst) => sum + inst.connections, 0),
      averageConnectionsPerInstance: Array.from(this.instances.values())
        .reduce((sum, inst) => sum + inst.connections, 0) / this.instances.size
    };
  }

  async _stopInstanceServer(instanceId) { /* ... */ }
  async _closeInstanceConnections(instanceId) { /* ... */ }
}
```

**Key Class: LoadBalancer**

```javascript
class LoadBalancer {
  constructor(poolManager, options = {}) {
    this.poolManager = poolManager;
    this.strategy = options.strategy || 'round-robin'; // round-robin, least-busy, sticky
    this.roundRobinIndex = 0;
    
    // Sticky session tracking (sessionId → instanceId)
    this.sessionMap = new Map();
    this.sessionTTL = options.sessionTTL || 3600000; // 1 hour
    
    // Health tracking
    this.instanceHealth = new Map();
  }

  /**
   * Route request to instance
   * @param {Object} request - Request to route
   * @returns {string} Instance ID
   */
  routeRequest(request) {
    // Check for sticky session
    if (request.sessionId) {
      const cachedInstance = this.sessionMap.get(request.sessionId);
      if (cachedInstance && this._isInstanceHealthy(cachedInstance)) {
        return cachedInstance;
      }
      // Clear stale mapping
      this.sessionMap.delete(request.sessionId);
    }
    
    // Select instance based on strategy
    let instanceId;
    switch (this.strategy) {
      case 'round-robin':
        instanceId = this._roundRobinSelect();
        break;
      case 'least-busy':
        instanceId = this._leastBusySelect();
        break;
      case 'custom':
        instanceId = this._customSelect(request);
        break;
      default:
        instanceId = this._roundRobinSelect();
    }
    
    // Cache sticky session
    if (request.sessionId) {
      this.sessionMap.set(request.sessionId, instanceId);
      
      // Schedule cleanup
      setTimeout(() => this.sessionMap.delete(request.sessionId), this.sessionTTL);
    }
    
    return instanceId;
  }

  /**
   * Handle failover - find alternative instance
   * @param {string} failedInstanceId - Failed instance
   * @param {Object} request - Original request
   * @returns {string|null} Alternative instance or null
   */
  failoverRequest(failedInstanceId, request) {
    const alternatives = this.poolManager.listInstances()
      .filter(inst => inst.id !== failedInstanceId && this._isInstanceHealthy(inst.id))
      .sort((a, b) => a.connections - b.connections);
    
    if (alternatives.length === 0) {
      return null;
    }
    
    return alternatives[0].id;
  }

  /**
   * Round-robin selection
   * @private
   */
  _roundRobinSelect() {
    const instances = this.poolManager.listInstances()
      .filter(inst => this._isInstanceHealthy(inst.id));
    
    if (instances.length === 0) {
      throw new Error('No healthy instances available');
    }
    
    const instance = instances[this.roundRobinIndex % instances.length];
    this.roundRobinIndex++;
    
    return instance.id;
  }

  /**
   * Least-busy selection (fewest connections)
   * @private
   */
  _leastBusySelect() {
    const instances = this.poolManager.listInstances()
      .filter(inst => this._isInstanceHealthy(inst.id))
      .sort((a, b) => a.connections - b.connections);
    
    if (instances.length === 0) {
      throw new Error('No healthy instances available');
    }
    
    return instances[0].id;
  }

  /**
   * Custom selection based on request properties
   * @private
   */
  _customSelect(request) {
    // Can be overridden for custom logic
    return this._leastBusySelect();
  }

  /**
   * Check instance health
   * @private
   */
  _isInstanceHealthy(instanceId) {
    const instance = this.poolManager.getInstanceState(instanceId);
    return instance && instance.status === 'healthy';
  }

  /**
   * Mark instance as degraded
   */
  markInstanceDegraded(instanceId) {
    const instance = this.poolManager.getInstanceState(instanceId);
    if (instance) {
      instance.status = 'degraded';
      console.warn(`[LoadBalancer] Instance ${instanceId} marked degraded`);
    }
  }

  /**
   * Update instance metrics
   */
  updateInstanceMetrics(instanceId, metrics) {
    const instance = this.poolManager.getInstanceState(instanceId);
    if (instance) {
      instance.metrics = metrics;
    }
  }

  /**
   * Get load distribution report
   * @returns {Object} Instance ID → load percentage
   */
  getLoadDistribution() {
    const instances = this.poolManager.listInstances();
    const totalConnections = instances.reduce((sum, i) => sum + i.connections, 0);
    
    const distribution = {};
    for (const inst of instances) {
      distribution[inst.id] = totalConnections > 0 
        ? ((inst.connections / totalConnections) * 100).toFixed(2) + '%'
        : '0%';
    }
    
    return distribution;
  }
}
```

```javascript
class ResourceMonitor {
  constructor(poolManager, options = {}) {
    this.poolManager = poolManager;
    this.checkInterval = options.checkInterval || 5000; // 5 sec
    
    // Thresholds
    this.cpuThreshold = options.cpuThreshold || 80;        // % CPU
    this.memoryThreshold = options.memoryThreshold || 85;  // % Memory
    this.connectionThreshold = options.connectionThreshold || 180; // connections
    
    // Historical data for trending
    this.history = new Map(); // instanceId → HistoricalData[]
    this.maxHistoryPoints = options.maxHistoryPoints || 60; // 60 samples
    
    // Start monitoring
    this._startMonitoring();
  }

  /**
   * Track CPU usage for instance
   */
  trackCPU(instanceId, cpuUsage) {
    this._recordMetric(instanceId, 'cpu', cpuUsage);
  }

  /**
   * Track memory usage for instance
   */
  trackMemory(instanceId, memoryUsage) {
    this._recordMetric(instanceId, 'memory', memoryUsage);
  }

  /**
   * Track connection count
   */
  trackConnections(instanceId, count) {
    this._recordMetric(instanceId, 'connections', count);
  }

  /**
   * Detect resource exhaustion
   * @returns {bool} True if exhausted
   */
  detectExhaustion(instanceId) {
    const instance = this.poolManager.getInstanceState(instanceId);
    if (!instance) return false;
    
    return instance.cpuUsage > this.cpuThreshold ||
           instance.memoryUsage > this.memoryThreshold ||
           instance.connections > this.connectionThreshold;
  }

  /**
   * Get resource report for all instances
   * @returns {ResourceReport}
   */
  getResourceReport() {
    const instances = this.poolManager.listInstances();
    const report = {
      timestamp: Date.now(),
      instances: {},
      aggregates: {
        totalCPU: 0,
        totalMemory: 0,
        totalConnections: 0,
        exhaustedInstances: []
      }
    };
    
    for (const inst of instances) {
      report.instances[inst.id] = {
        cpu: inst.cpuUsage,
        memory: inst.memoryUsage,
        connections: inst.connections,
        exhausted: this.detectExhaustion(inst.id)
      };
      
      report.aggregates.totalCPU += inst.cpuUsage;
      report.aggregates.totalMemory += inst.memoryUsage;
      report.aggregates.totalConnections += inst.connections;
      
      if (this.detectExhaustion(inst.id)) {
        report.aggregates.exhaustedInstances.push(inst.id);
      }
    }
    
    const count = instances.length;
    if (count > 0) {
      report.aggregates.avgCPU = (report.aggregates.totalCPU / count).toFixed(2);
      report.aggregates.avgMemory = (report.aggregates.totalMemory / count).toFixed(2);
    }
    
    return report;
  }

  /**
   * Get metrics for specific instance
   */
  getMetrics(instanceId) {
    const historicalData = this.history.get(instanceId) || [];
    
    if (historicalData.length === 0) {
      return null;
    }
    
    // Calculate statistics
    const cpuValues = historicalData.map(d => d.cpu);
    const memoryValues = historicalData.map(d => d.memory);
    const connectionValues = historicalData.map(d => d.connections);
    
    return {
      cpu: {
        current: cpuValues[cpuValues.length - 1],
        avg: this._average(cpuValues),
        min: Math.min(...cpuValues),
        max: Math.max(...cpuValues)
      },
      memory: {
        current: memoryValues[memoryValues.length - 1],
        avg: this._average(memoryValues),
        min: Math.min(...memoryValues),
        max: Math.max(...memoryValues)
      },
      connections: {
        current: connectionValues[connectionValues.length - 1],
        avg: this._average(connectionValues),
        min: Math.min(...connectionValues),
        max: Math.max(...connectionValues)
      }
    };
  }

  /**
   * Record metric data point
   * @private
   */
  _recordMetric(instanceId, metric, value) {
    if (!this.history.has(instanceId)) {
      this.history.set(instanceId, []);
    }
    
    const data = this.history.get(instanceId);
    const latest = data[data.length - 1] || { cpu: 0, memory: 0, connections: 0 };
    
    const updated = { ...latest, [metric]: value };
    data.push(updated);
    
    // Keep only recent history
    if (data.length > this.maxHistoryPoints) {
      data.shift();
    }
  }

  /**
   * Start monitoring loop
   * @private
   */
  _startMonitoring() {
    this.monitoringTimer = setInterval(() => {
      const report = this.getResourceReport();
      
      // Log if instances are exhausted
      if (report.aggregates.exhaustedInstances.length > 0) {
        console.warn(`[ResourceMonitor] Exhausted instances: ${report.aggregates.exhaustedInstances.join(', ')}`);
      }
    }, this.checkInterval);
  }

  /**
   * Calculate average
   * @private
   */
  _average(values) {
    if (values.length === 0) return 0;
    return (values.reduce((a, b) => a + b, 0) / values.length).toFixed(2);
  }

  /**
   * Trigger auto-scale
   */
  triggerAutoScale(direction) {
    if (direction === 'up') {
      console.log('[ResourceMonitor] Triggering scale-up');
      this.poolManager.spinUpInstance();
    } else if (direction === 'down') {
      console.log('[ResourceMonitor] Triggering scale-down');
      const idleInstance = this.poolManager._findIdleInstance();
      if (idleInstance) {
        this.poolManager.spinDownInstance(idleInstance.id);
      }
    }
  }

  /**
   * Cleanup
   */
  destroy() {
    if (this.monitoringTimer) {
      clearInterval(this.monitoringTimer);
    }
  }
}
```

```javascript
class DistributedSessionCoordinator {
  constructor(options = {}) {
    // Use Redis or in-memory cache
    this.useRedis = options.useRedis || false;
    this.redisClient = options.redisClient || null;

    this.sessionCache = new Map();
    this.sessionTTL = options.sessionTTL || 3600000; // 1 hour

  /**
   * Store session data in distributed cache
   */
  async set(sessionId, data, ttl = null) {
    ttl = ttl || this.sessionTTL;
    
    try {
      if (this.useRedis && this.redisClient) {
        await this.redisClient.setex(
          `session:${sessionId}`,
          Math.ceil(ttl / 1000),
          JSON.stringify(data)
        );
      } else {
        this.sessionCache.set(sessionId, data);
        
        // Schedule cleanup
        setTimeout(() => this.sessionCache.delete(sessionId), ttl);
      }
    } catch (error) {
      console.error(`[DistributedSessionCoordinator] Failed to store session ${sessionId}: ${error.message}`);
      throw error;
    }
  }

  /**
   * Retrieve session data
   */
  async get(sessionId) {
    try {
      if (this.useRedis && this.redisClient) {
        const data = await this.redisClient.get(`session:${sessionId}`);
        return data ? JSON.parse(data) : null;
      } else {
        return this.sessionCache.get(sessionId) || null;
      }
    } catch (error) {
      console.error(`[DistributedSessionCoordinator] Failed to retrieve session ${sessionId}: ${error.message}`);
      return null;
    }
  }

  /**
   * Atomically update session data
   */
  async setAtomic(sessionId, updateFn) {
    const current = await this.get(sessionId) || {};
    const updated = updateFn(current);
    await this.set(sessionId, updated);
    return updated;
  }

  /**
   * Initiate session failover
   * @param {string} sessionId - Session to failover
   * @param {string} failedInstanceId - Failed instance
   * @param {string} targetInstanceId - Target instance
   * @returns {Promise<FailoverResult>}
   */
  async initiateFailover(sessionId, failedInstanceId, targetInstanceId) {
    console.log(`[DistributedSessionCoordinator] Initiating failover for session ${sessionId}`);
    
    const startTime = Date.now();
    
    try {
      // Step 1: Retrieve session state from cache
      const sessionState = await this.get(sessionId);
      if (!sessionState) {
        throw new Error(`Session ${sessionId} not found in cache`);
      }
      
      // Step 2: Restore session on target instance
      await this._restoreSessionOnInstance(sessionId, targetInstanceId, sessionState);
      
      // Step 3: Validate migration
      const validationResult = await this._validateMigration(sessionId, targetInstanceId);
      if (!validationResult.success) {
        throw new Error(`Validation failed: ${validationResult.error}`);
      }
      
      // Step 4: Clear failed instance cache
      await this.delete(sessionId);
      
      // Record success
      const failoverTime = Date.now() - startTime;
      
      console.log(`[DistributedSessionCoordinator] Failover completed in ${failoverTime}ms`);
      
      return {
        success: true,
        sessionId,
        fromInstance: failedInstanceId,
        toInstance: targetInstanceId,
        failoverTime,
        timestamp: Date.now()
      };
    } catch (error) {
      console.error(`[DistributedSessionCoordinator] Failover failed: ${error.message}`);
      
      // Attempt rollback
      await this._rollbackFailover(sessionId);
      
      return {
        success: false,
        sessionId,
        error: error.message,
        timestamp: Date.now()
      };
    }
  }

  /**
   * Auto-recover on disconnect
   */
  async autoRecoverOnDisconnect(sessionId) {
    console.log(`[DistributedSessionCoordinator] Attempting auto-recovery for ${sessionId}`);
    
    try {
      const sessionState = await this.get(sessionId);
      if (!sessionState) {
        return false;
      }
      
      // Session data available, recovery possible
      return true;
    } catch (error) {
      console.error(`[DistributedSessionCoordinator] Recovery check failed: ${error.message}`);
      return false;
    }
  }

  /**
   * Resolve conflicting session states
   */
  async resolveConflict(sessionId, state1, state2) {
    // Simple strategy: newer timestamp wins
    const timestamp1 = state1._lastUpdated || 0;
    const timestamp2 = state2._lastUpdated || 0;
    
    return timestamp1 >= timestamp2 ? state1 : state2;
  }

  /**

  async mergeSessionUpdates(updates) {

    for (const update of updates) {

    }

  }
```

## APPENDIX: GLOSSARY

**Pool:** Set of managed browser instances
**Instance:** Single Electron browser with WebSocket server
**Load Balancer:** Routes requests to instances based on strategy
**Failover:** Automatic migration of session to healthy instance
**Session Coordinator:** Manages distributed session state
**Sticky Session:** Route same client to same instance
**Round-Robin:** Distribute requests equally across instances
**Least-Busy:** Route to instance with fewest connections
**Resource Monitor:** Tracks CPU, memory, connections
**Auto-Scale:** Automatically add/remove instances based on load
