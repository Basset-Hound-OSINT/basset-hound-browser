2. Reputation-Based Proxy Selection:

const ReputationScorer = require('./reputation-scorer');
const scorer = new ReputationScorer();

// Register proxy
scorer.registerProxyForScoring('proxy-1');

// Record metrics
scorer.updateProxyMetrics('proxy-1', {
  success: true,
  latency: 100,
  userAgentAccepted: true
});

// Get health status
const report = scorer.getProxyReport('proxy-1');
if (report.status === 'healthy') {
  useProxy('proxy-1');
} else if (report.status === 'excluded') {
  switchToFallback();
}

3. Intelligent Fallback on Provider Outage:

const FallbackStrategy = require('./fallback-strategy');
const fallback = new FallbackStrategy();

// Initialize
fallback.initializeFallbackPath('session-1', 'primary-proxy');

// Register alternatives
fallback.registerProxyForFallback({
  proxyId: 'fallback-1',
  region: 'US',
  provider: 'Oxylabs'
});

// Detect failure and fallback
if (requestFailed) {
  const result = fallback.determineFallback('session-1', 'primary-proxy', 'blocked');
  if (result.fallback) {
    const newProxy = result.newProxy;
    switchProxy(newProxy);
  }
}
