USAGE EXAMPLES

1. QUEUE MANAGER:
   const queueManager = new QueueManager({ poolSize: 10 });
   await queueManager.connect();
   await queueManager.declareQueue('monitoring:tasks');
   const result = await queueManager.publishMessage('monitoring:tasks', {
     type: 'MONITORING_CHECK',
     payload: { url: 'https://example.com' }
   });

2. MESSAGE HANDLER:
   const handler = new MessageHandler({ maxConcurrency: 10 });
   handler.registerHandler('monitoring:tasks', async (message) => {
     // Process message
     return { processed: true };
   });

3. STREAM PROCESSOR:
   const processor = new StreamProcessor({ windowSize: 300000 });
   await processor.start();
   processor.registerAlertRule('critical', {
     name: 'Critical Alert',
     condition: 'HIGH_SEVERITY',
     actions: ['notify']
   });
   await processor.processEvent({ event_id: '1', task_id: 'task_1', ... });

4. EVENT ROUTER:
   const router = new EventRouter();
   router.registerRoute('route_1', {
     pattern: 'alert',
     destination: 'handler'
   });
   router.registerHandler('handler', async (event) => {
     return { delivered: true };
   });
   const result = await router.routeEvent({ type: 'alert', ... });

5. TASK SCHEDULER:
   const scheduler = new TaskScheduler();
   scheduler.defineTask('REPORT_GEN', {
     handler: async (payload) => ({ report: true })
   });
   scheduler.scheduleTask('task_1', 'REPORT_GEN', {}, {
     scheduledTime: Date.now(),
     recurrence: 'DAILY'
   });

6. BACKGROUND JOBS:
   const jobs = new BackgroundJobs();
   const job = jobs.createJob('report_generation', {
     reportType: 'MONITORING'
   }, { priority: 'high' });
